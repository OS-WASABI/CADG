344 // Attempts to get the most recent ODBC error as a string.
345 // Always returns std::string, even in unicode mode.
346 inline std::string
347 recent_error(SQLHANDLE handle, SQLSMALLINT handle_type, long& native, std::string& state)
348 {
349  nanodbc::string result;
350  std::string rvalue;
351  std::vector<NANODBC_SQLCHAR> sql_message(SQL_MAX_MESSAGE_LENGTH);
352  sql_message[0] = '\0';
353 
354  SQLINTEGER i = 1;
355  SQLINTEGER native_error = 0;
356  SQLSMALLINT total_bytes = 0;
357  NANODBC_SQLCHAR sql_state[6] = {0};
358  RETCODE rc;
359 
360  do
361  {
362  NANODBC_CALL_RC(
363  NANODBC_FUNC(SQLGetDiagRec),
364  rc,
365  handle_type,
366  handle,
367  (SQLSMALLINT)i,
368  sql_state,
369  &native_error,
370  0,
371  0,
372  &total_bytes);
373 
374  if (success(rc) && total_bytes > 0)
375  sql_message.resize(static_cast<std::size_t>(total_bytes) + 1);
376 
377  if (rc == SQL_NO_DATA)
378  break;
379 
380  NANODBC_CALL_RC(
381  NANODBC_FUNC(SQLGetDiagRec),
382  rc,
383  handle_type,
384  handle,
385  (SQLSMALLINT)i,
386  sql_state,
387  &native_error,
388  sql_message.data(),
389  (SQLSMALLINT)sql_message.size(),
390  &total_bytes);
391 
392  if (!success(rc))
393  {
394  convert(std::move(result), rvalue);
395  return rvalue;
396  }
397 
398  if (!result.empty())
399  result += ' ';
400 
401  result += nanodbc::string(sql_message.begin(), sql_message.end());
402  i++;
403 
404 // NOTE: unixODBC using PostgreSQL and SQLite drivers crash if you call SQLGetDiagRec()
405 // more than once. So as a (terrible but the best possible) workaround just exit
406 // this loop early on non-Windows systems.
407 #ifndef _MSC_VER
408  break;
409 #endif
410  } while (rc != SQL_NO_DATA);
411 
412  convert(std::move(result), rvalue);
413  if (size(sql_state) > 0)
414  {
415  state.clear();
416  state.reserve(size(sql_state) - 1);
417  for (std::size_t idx = 0; idx != size(sql_state) - 1; ++idx)
418  {
419  state.push_back(static_cast<char>(sql_state[idx]));
420  }
421  }
422 
423  native = native_error;
424  std::string status = state;
425  status += ": ";
426  status += rvalue;
427 
428  // some drivers insert \0 into error messages for unknown reasons
429  using std::replace;
430  replace(status.begin(), status.end(), '\0', ' ');
431 
432  return status;
433 }
